<template>
    <div class="datepicker">
        <input type="text" name="datepicker"/>
        <div class="modal">
            <template v-if="range">
                <Calendar :year="_year" 
                          :month="_month" 
                          :onEndSelect="()=>{}" 
                          :onStartSelect="()=>{}" 
                          disableNext/>
                <Calendar :year="_year" 
                          :month="_month + 1" 
                          :onEndSelect="()=>{}" 
                          :onStartSelect="()=>{}" 
                          disablePrev/>
            </template>
            <template v-else>
                <Calendar :onEndSelect="()=>{}" :onStartSelect="()=>{}"/>
            </template>
        </div>
     </div>
</template>

<script>
import Calendar from './Calendar.vue'

export default {
    name: 'DatePicker',
    components: {
        Calendar
    },
    data: function() {
        return {
            selected: this.initState ? Array.from(this.initState) : [],
            state: this.initState ? this.selected[0] : new Date(),
        }
    },
    methods: {
        
    },
    computed: {
        _year: function() {
            return this.state.getFullYear();
        },
        _month: function() {
            return this.state.getMonth();
        }
    },
    props: {
        range: Boolean,
        initState: [Array, Date]
    }
}
</script>

<style scoped>
    .datepicker {
        display: inline-block;
    }
</style>
